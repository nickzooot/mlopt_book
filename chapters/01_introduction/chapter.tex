\begin{abstract}
We summarize first-order and second-order conditions for local minima, 
recall differential notation, discuss error sources in numerical 
differentiation, and collect basic convexity, strong convexity, and 
Lipschitz smoothness facts that will be used throughout the course.
\end{abstract}

\section{Optimization problem}
We study unconstrained optimization problems of the form
\[
f(x) \rightarrow \min _{x},
\]
with the practical goal of finding good local minima. gradients and 
hessians are denoted by
\[
\nabla f(x)=\left[\begin{array}{c}
\frac{\partial f}{\partial x_{1}} \\
\vdots \\
\frac{\partial f}{\partial x_{n}}
\end{array}\right] \in \mathbb{R}^{n}, \qquad
\nabla^{2} f(x)=\left\{\frac{\partial^{2} f}{\partial x_{i} \partial x_{j}}\right\}_{i, j=1}^{n} \in \mathbb{R}^{n \times n} .
\]

    \centering

   \caption{left: rugged nonconvex loss with many minima and a sharp global minimum (red). Right: sharp global minimum and flatter local minimum (red).}\label{fig:}

\hfill
\noindent Many ML losses are highly nonconvex, with numerous sharp minima; the sharp global minimum can be sensitive to initialization and noise, while flatter local minima often yield more robust solutions even if their objective value is slightly higher.

\section{Local optimality conditions}

\begin{claim}[Necessary condition of a local minimum]
If $f \in C^{1}$, $x \in \operatorname{int}(\operatorname{dom} f)$, and $x$ is a local minimum, then $\nabla f(x)=0$. If in addition $f \in C^{2}$, then $\nabla^{2} f(x) \succeq 0$.
\end{claim}

\begin{claim}[Sufficient condition of a local minimum]
If $f \in C^{2}$, $x \in \operatorname{int}(\operatorname{dom} f)$, $\nabla f(x)=0$, and $\nabla^{2} f(x) \succ 0$, then $x$ is a local minimum.
\end{claim}

\section{Differentials}

\begin{definition}
The function $f$ is differentiable at $x$ if
\[
f(x+h)-f(x)=\mathrm{d} f(x)[h]+o(\|h\|),
\]
where $\mathrm{d} f(x)$ is a linear operator.
\end{definition}

\begin{definition}
The second differential is
\[
\mathrm{d}^{2} f(x)\left[h_{1}, h_{2}\right]=\mathrm{d}\left(\mathrm{d} f(x)\left[h_{1}\right]\right)(x)\left[h_{2}\right].
\]
\end{definition}

\section{Error analysis and machine precision}

For floating-point arithmetic with machine precision $\varepsilon_{m}$, relative errors satisfy
\[
\frac{|f(x)-x|}{|x|} \leq \varepsilon_{m}.
\]
Typical values:
\[
\begin{array}{c|c}
\text { bits } & \varepsilon_{m} \\
\hline
64 & 10^{-16} \\
32 & 10^{-7} \\
16 & 10^{-3}
\end{array}
\]

\section{Convexity and strong convexity}

\begin{claim}[Differential criteria of convexity]
Let $f \in C^{1}$. Then $f$ is convex if and only if for all $x, y$,
\[
f(y) \geq f(x)+\nabla f(x)^{\top}(y-x) .
\]
If $f \in C^{2}$, then $f$ is convex if and only if $\nabla^{2} f(x) \succeq 0$ for all $x$.
\end{claim}

\begin{definition}
The function $f$ is $\mu$-strongly convex if for all $x$ the function $x \mapsto f(x)-\frac{\mu}{2}\|x\|^{2}$ is convex.
\end{definition}

\begin{claim}[Differential criteria of strong convexity]
If $f \in C^{1}$, then $f$ is $\mu$-strongly convex if and only if for all $x, y$,
\[
f(y) \geq f(x)+\nabla f(x)^{\top}(y-x)+\frac{\mu}{2}\|x-y\|^{2} .
\]
If $f \in C^{2}$, then $f$ is $\mu$-strongly convex if and only if $\nabla^{2} f(x) \succeq \mu I \succ 0$ for all $x$.
\end{claim}

\section{Lipschitz classes}

\begin{definition}
The function $f$ belongs to the Lipschitz class $C_{L}^{k, m}$ if (i) $f \in C^{k}$, (ii) $m \leq k$, and (iii)
\[
\left\|\nabla^{m} f(x)-\nabla^{m} f(y)\right\| \leq L\|x-y\| \quad \forall x, y .
\]
\end{definition}

\paragraph{Examples.}
\begin{enumerate}
\item If $f \in C_{L}^{0,0}$, then $|f(x)-f(y)| \leq L\|x-y\|$ for all $x, y$, equivalently
\[
f(y)-L\|x-y\| \leq f(x) \leq f(y)+L\|x-y\| .
\]
\item If $f \in C_{L}^{1,1}$, then
\[
f(x) \geq f(y)+\nabla f(y)^{\top}(x-y)-\frac{L}{2}\|x-y\|^{2},
\]
\[
f(x) \leq f(y)+\nabla f(y)^{\top}(x-y)+\frac{L}{2}\|x-y\|^{2}.
\]
\end{enumerate}

\begin{claim}
If $f \in C_{L}^{k, m-1}$, then $\|\nabla^{m} f(x)\| \leq L$.
\end{claim}

\section{Finite differentiation}

We approximate directional derivatives $\nabla f(x)^{\top} d$ with stepsize $\varepsilon$ and machine precision $\varepsilon_{m}$.

\paragraph{Forward difference.} Using
\[
f(x+\varepsilon d)=f(x)+\nabla f(x)^{\top} \varepsilon d+O\left(\varepsilon^{2}\right),
\]
the estimator
\[
f_{1}(\varepsilon)=\frac{f(x+\varepsilon d)-f(x)}{\varepsilon}=\nabla f(x)^{\top} d+O(\varepsilon)
\]
has error bounded by
\[
\left|f_{1}(\varepsilon)-\nabla f(x)^{\top} d\right| \leq L_{1} \varepsilon+\frac{L_{2}}{\varepsilon} \varepsilon_{m},
\]
so the optimal $\varepsilon$ balances truncation and roundoff: $\varepsilon=\sqrt{\varepsilon_{m}}$, giving $\left|f_{1}(\varepsilon)\right| \lesssim C \sqrt{\varepsilon_{m}}$.

\paragraph{Central difference.} With
\[
f(x+\varepsilon d)=f(x)+\nabla f(x)^{\top} \varepsilon d+\frac{\varepsilon^{2}}{2} d^{\top} \nabla^{2} f(x) d+O\left(\varepsilon^{3}\right),
\]
\[
f(x-\varepsilon d)=f(x)-\nabla f(x)^{\top} \varepsilon d+\frac{\varepsilon^{2}}{2} d^{\top} \nabla^{2} f(x) d+O\left(\varepsilon^{3}\right),
\]
the estimator
\[
\frac{f(x+\varepsilon d)-f(x-\varepsilon d)}{2 \varepsilon}=\nabla f(x)^{\top} d+O\left(\varepsilon^{2}\right)
\]
has error bounded by
\[
L_{0} \varepsilon^{2}+\frac{L_{1}}{\varepsilon} \varepsilon_{m},
\]
optimized at $\varepsilon \approx\left(\varepsilon_{m}\right)^{1 / 3}$.

\paragraph{Complex step.} Using the expansion
\[
f(x+i \varepsilon d)=f(x)+i \varepsilon \nabla f(x)^{\top} d+O\left(\varepsilon^{2}\right)+i O\left(\varepsilon^{3}\right),
\]
we obtain
\[
\frac{\operatorname{Im} f(x+i \varepsilon d)}{\varepsilon}+\!O\left(\varepsilon^{2}\right)=\nabla f(x)^{\top} d .
\]
Because there is no subtraction, instability from $a \approx b$ is avoided, so one can take $\varepsilon=\varepsilon_{m}$.
