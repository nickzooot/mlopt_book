\section{Optimization problem}
We will start our course with a study of unconstrained optimization problems 
of the form
\[
  f(\mathbf{x}) \rightarrow \min_{\mathbf{x} \in \mathbb{R}^{n}},
\]
with the practical goal of finding minima. Gradients and Hessians
are denoted by
\[
\nabla f(\mathbf{x})=\begin{pmatrix} \frac{\partial f}{\partial x_1}\\ \vdots\\
\frac{\partial f}{\partial x_n} \end{pmatrix} \in \mathbb{R}^{n }, \qquad
\nabla^{2} f(\mathbf{x})=\left\{\frac{\partial^{2} f}{\partial x_{i}\partial x_{j}}
\right\}_{i, j=1}^{n} \in \mathbb{R}^{n \times n} .
\]

\begin{figure}[ht]
    \centering
    \incfig{global-and-flat-local-minima}
    \caption{Rugged nonconvex loss with many minima and a sharp 
    global minimum.}
    \label{fig:global-and-flat-local-minima}
\end{figure}

%\hfill
\noindent Many ML losses are highly nonconvex, with numerous sharp minima
(see \cref{fig:global-and-flat-local-minima}). The sharp global minimum can make the final
model sensitive to inputs and noise, while flatter local minima often yield more robust
solutions even if their objective value is slightly higher. In general, finding a good
local minimum can be more valuable than finding the global minimum. Moreover, ML
problems are typically high-dimensional, which makes finding a global minimum very
complicated, if it is even viable. Therefore, our first goal is not necessarily to rush
toward the global minimum, but to find a good local minimum: this is often easier and
can lead to better results. We start by reviewing basic optimality conditions for local minima.

\section{Local optimality conditions}
In many problems, the first step is to write down optimality 
conditions: in simple cases they can lead to a closed-form solution, 
and in iterative algorithms they often serve as stopping criteria.

\begin{definition}[Local minimum]
A point $\mathbf{x}$ is a (non-strict) local minimum of $f$ if there exists
$\varepsilon>0$ such that

\[
  f(\mathbf{y})\ge f(\mathbf{x})\quad \forall \mathbf{y}\in
  O_\varepsilon(\mathbf{x}):=\{\mathbf{y}:\|\mathbf{y}-\mathbf{x}\|<\varepsilon\}.
\]

If the inequality is strict for all $\mathbf{y}\neq \mathbf{x}$ in
$O_\varepsilon(\mathbf{x})$, then $\mathbf{x}$ is a strict local minimum.
\end{definition}


\begin{claim}[Necessary condition of a local minimum]
If $f \in C^{1}$, $\mathbf{x} \in \operatorname{int}(\operatorname{dom} f)$,
and $\mathbf{x}$ is a local minimum, then $\nabla f(\mathbf{x})=0$. If in
addition $f \in C^{2}$, then $\nabla^{2} f(\mathbf{x}) \succeq 0$.
\end{claim}
\begin{proof}
Let $\mathbf{d}\in\mathbb{R}^n$ be arbitrary and define the one-dimensional function
\[
  \varphi(\alpha)=f(\mathbf{x}+\alpha \mathbf{d}).
\]
Since $\mathbf{x}$ is a local minimum and $\mathbf{x}\in \operatorname{int}
(\operatorname{dom} f)$, for sufficiently small $|\alpha|$ we have
$\mathbf{x}+\alpha \mathbf{d}\in \operatorname{dom} f$ and $\varphi(\alpha)\ge \varphi(0)$,
so $\alpha=0$ is a local minimum of $\varphi$.
Because $f\in C^1$, we can use Taylor expansion:
\[
\varphi(\alpha) = \varphi(0) + \varphi'(0)\alpha + o(\alpha) \ge \varphi(0)
\quad \overset{:/\alpha}{\Longrightarrow}\quad \varphi'(0)+o(1)\ge 0
\quad \overset{\alpha\to 0}{\Longrightarrow}\quad \varphi'(0)\ge 0.
\]
Hence, we have $\varphi'(0) = \nabla f(\mathbf{x})^{\top} \mathbf{d} \ge 0$ for all $\mathbf{d}$.
This means that if we take $\mathbf{d}$ and $-\mathbf{d}$, the inequality can hold only if
$\nabla f(\mathbf{x}) = 0$.
If in addition $f\in C^2$, then for any $d$ we have the second-order 
expansion

\[
  f(\mathbf{x}+\alpha \mathbf{d})=f(\mathbf{x})+\alpha\nabla f(\mathbf{x})^\top \mathbf{d}
  +\frac{\alpha^2}{2}\mathbf{d}^\top\nabla^2 f(\mathbf{x})\,\mathbf{d}+o(\alpha^2).
\]

Using $\nabla f(\mathbf{x})=0$ and the same logic as above, we obtain that
$\mathbf{d}^\top\nabla^2 f(\mathbf{x})\,\mathbf{d}\ge 0$ for all $\mathbf{d}$,
i.e.\ $\nabla^2 f(\mathbf{x})\succeq 0$.
\end{proof}

\begin{figure}[ht]
    \centering
    \incfig{why-just-necessary}
    \caption{Explanation of why $\nabla^2 f(\mathbf{x})\succeq 0$ and $\nabla f(\mathbf{x})=0$
    are only necessary conditions.}
    \label{fig:why-just-necessary}
\end{figure}

\noindent\textbf{Why only necessary?}

\noindent A stationary point with $\nabla f(\mathbf{x})=0$ can be a local minimum,
a local maximum, or a saddle point.
The condition $\nabla^2 f(\mathbf{x})\succeq 0$ rules out strict local maxima,
but still allows saddle points when the Hessian is only 
semidefinite (see \Cref{fig:why-just-necessary}).

\begin{claim}[Sufficient condition of a local minimum]
If $f \in C^{2}$, $\mathbf{x} \in \operatorname{int}(\operatorname{dom} f)$,
$\nabla f(\mathbf{x})=0$, and $\nabla^{2} f(\mathbf{x}) \succ 0$, then
$\mathbf{x}$ is a local minimum.
\end{claim}
\begin{proof}
By the second-order Taylor expansion,
\[
  f(\mathbf{x}+\mathbf{h})=f(\mathbf{x})+\nabla f(\mathbf{x})^\top \mathbf{h}
  +\frac{1}{2}\mathbf{h}^\top\nabla^2 f(\mathbf{x})\,
  \mathbf{h}+o(\|\mathbf{h}\|^2).
\]
If $\nabla f(\mathbf{x})=0$ and $\nabla^2 f(\mathbf{x})\succ 0$, then
$\mathbf{h}^\top\nabla^2 f(\mathbf{x})\,\mathbf{h}>0$ for all $\mathbf{h}\ne 0$.
Therefore, for sufficiently small $\mathbf{h}\ne 0$ we have
$f(\mathbf{x}+\mathbf{h})>f(\mathbf{x})$.
\end{proof}

\section{Differentials}

\begin{definition}
The function $f$ is differentiable at $\mathbf{x}$ if
\[
f(\mathbf{x}+\mathbf{h})-f(\mathbf{x})=\mathrm{d} f(\mathbf{x})[\mathbf{h}]+o(\|\mathbf{h}\|),
\]
where $\mathrm{d} f(\mathbf{x})$ is a linear operator.
\end{definition}

\begin{definition}
The second differential is
\[
\mathrm{d}^{2} f(\mathbf{x})\left[\mathbf{h}_{1}, \mathbf{h}_{2}\right]=
\mathrm{d}\left(\mathrm{d} f(\mathbf{x})\left[\mathbf{h}_{1}\right]\right)(\mathbf{x})
\left[\mathbf{h}_{2}\right].
\]
\end{definition}

\noindent For a scalar function $f:\mathbb{R}^n\to\mathbb{R}$, 
the first differential is the scalar product with the gradient:
\[
  \mathrm{d} f(\mathbf{x})[\mathbf{h}]=\nabla f(\mathbf{x})^\top \mathbf{h}.
\]
The second differential is a bilinear form linked to the Hessian:
\[
  \mathrm{d}^2 f(\mathbf{x})[\mathbf{h}_1,\mathbf{h}_2]=
  \mathbf{h}_1^\top \nabla^2 f(\mathbf{x})\,\mathbf{h}_2.
\]

\begin{example}[A worked example: $f(\mathbf{x})=\frac{1}{3}\|\mathbf{x}\|_2^3$]
Let $f(\mathbf{x})=\frac{1}{3}\|\mathbf{x}\|_2^3=\frac{1}{3}(\mathbf{x}^\top \mathbf{x})^{3/2}$.
For $\mathbf{x}\neq 0$,
\[
  \nabla f(\mathbf{x})=\|\mathbf{x}\|_2\,\mathbf{x},
  \qquad
  \nabla^2 f(\mathbf{x})=\frac{1}{\|\mathbf{x}\|_2}\mathbf{x}\mathbf{x}^\top
  +\|\mathbf{x}\|_2 \mathbf{I}.
\]
\end{example}

\section{Error analysis and machine precision}

In the algorithms covered in this book, we use only real numbers. However, modern
computers cannot store and manipulate arbitrary real numbers. Instead, they work only
with a finite subset of numbers known as floating-point numbers. This means that
computations produce only approximations, so we want these approximations to be as
close to the theoretical values as possible. In floating-point arithmetic, a real number is
stored in the form $\mathrm{fl}(x)=sM2^E$. Machine precision $\varepsilon_m$ bounds the
relative representation error:
\[
\left|\frac{\mathrm{fl}(x)-x}{x}\right| \leq \varepsilon_m\qquad (x\neq 0).
\]
Typical values:
\[
\begin{array}{c|c}
\text{bits} & \varepsilon_m \\
\hline
64 & 10^{-16} \\
32 & 10^{-7} \\
16 & 10^{-3}
\end{array}
\]

\noindent In practice, this means that when we plot an error versus the number of
iterations, we should stop our algorithm once the error is on the order of $\varepsilon_m$,
because improvements beyond that are below machine precision (see \Cref{fig:finite-difference-tradeoff}).

    \caption{Finite-difference tradeoff in 64-bit computations.}
    \label{fig:finite-difference-tradeoff}
\end{figure}
\section{Classes of functions}

Unfortunately, Taylor expansions are useful only in a neighborhood of a point. To prove
convergence of optimization algorithms, we therefore need to introduce common classes
of functions whose defining properties hold globally rather than locally.
\subsection{Convexity and strong convexity}
\label{sec:ch1-convexity}

\begin{definition}[Convex function (see \Cref{fig:kinds-of-convexity})]
\label{def:ch1-convex-function}
A function $f$ is convex if for all $\mathbf{x},\mathbf{y}$ and $\alpha\in[0,1]$,
\[
  f(\alpha \mathbf{x}+(1-\alpha)\mathbf{y})
  \le \alpha f(\mathbf{x})+(1-\alpha)f(\mathbf{y}).
\]
It is strictly convex if the inequality is strict for $\mathbf{x}\ne \mathbf{y}$ and
$\alpha\in(0,1)$.\\
\noindent \textbf{Note:} A function is concave if $-f$ is convex (strictly concave if
$-f$ is strictly convex).
\end{definition}


\begin{claim}[Differential criteria of convexity]
\label{clm:ch1-diff-convexity}
Let $f \in C^{1}$. Then $f$ is convex if and only if for all $\mathbf{x}, \mathbf{y}$,
\[
f(\mathbf{y}) \geq f(\mathbf{x})+\nabla f(\mathbf{x})^{\top}(\mathbf{y}-\mathbf{x}) .
\]
If $f \in C^{2}$, then $f$ is convex if and only if $\nabla^{2} f(\mathbf{x}) 
\succeq 0$ for all $\mathbf{x}$ (see \Cref{fig:kinds-of-convexity}). 
\end{claim}

Geometrically, this claim means that the graph of the function must be no lower than
the tangent plane drawn to it at any point.


\begin{definition}[$\mu$-strongly convex]
The function $f$ is $\mu$-strongly convex if for all $\mathbf{x}$ the function 
$f(\mathbf{x})-\frac{\mu}{2}\|\mathbf{x}\|^{2}$ is convex.
\end{definition}

The geometric view can be better understood from the following claim.
\begin{claim}[Differential criteria of strong convexity]\label{clm:diff-strong-conv}
If $f \in C^{1}$, then $f$ is $\mu$-strongly convex if and only if for 
all $\mathbf{x}, \mathbf{y}$,
\[
f(\mathbf{y}) \geq f(\mathbf{x})+\nabla f(\mathbf{x})^{\top}(\mathbf{y}-\mathbf{x})
+\frac{\mu}{2}\|\mathbf{x}-\mathbf{y}\|^{2} .
\]
If $f \in C^{2}$, then $f$ is $\mu$-strongly convex if and only if
$\nabla^{2} f(\mathbf{x}) \succeq \mu \mathbf{I} \succ 0$ for all $\mathbf{x}$.\\
\noindent \textbf{Note:} This is also equivalent to $\lambda_{min}
(\nabla^2f(\mathbf{x} ))\ge \mu$
\end{claim}

\noindent From this claim, it is easy to see that if $f$ is $\mu$-strongly convex, then it
is convex. From a geometric perspective, the class of $\mu$-strongly convex functions
includes convex functions that are not too flat and have curvature bounded below (see \Cref{fig:kinds-of-convexity}).

    \centering

\subsection{Lipschitz classes}

Convexity describes the shape of $f$, but it does not quantify how rapidly the function (or
its derivatives) can vary. For gradient-based methods, such quantitative control is crucial:
it lets us bound Taylor remainders and obtain convergence rates. Lipschitz classes provide
a compact way to encode these smoothness features.

\begin{definition}[Lipschitz class $C_L^{k,m}$]
The function $f$ belongs to the Lipschitz class $C_{L}^{k, m}$ if (i) 
$f \in C^{k}$, (ii) $m \leq k$, and (iii)
\[
\left\|\nabla^{m} f(\mathbf{x})-\nabla^{m} f(\mathbf{y})\right\|
\leq L\|\mathbf{x}-\mathbf{y}\|
\quad \forall \mathbf{x}, \mathbf{y} .
\]
\end{definition}

Two special cases are used throughout the book: $C_L^{0,0}$ (Lipschitz continuity of the
function values) and $C_L^{1,1}$ (Lipschitz continuity of the gradient), also known as
$L$-smoothness. We now state some claims that we will use later; we omit the proofs (the
first one is straightforward and the second one is more complicated).
\begin{claim}
If $f \in C_{L}^{0,0}$, then $|f(\mathbf{x})-f(\mathbf{y})| \leq L\|\mathbf{x}-\mathbf{y}\|$
for all $\mathbf{x}, \mathbf{y}$,
  equivalently
\[
f(\mathbf{y})-L\|\mathbf{x}-\mathbf{y}\| \leq f(\mathbf{x})
\leq f(\mathbf{y})+L\|\mathbf{x}-\mathbf{y}\| .
\]
\end{claim}

\begin{claim}
If $f \in C_{L}^{1,1}$, then
\[
f(\mathbf{x}) \geq f(\mathbf{y})+\nabla f(\mathbf{y})^{\top}(\mathbf{x}-\mathbf{y})
-\frac{L}{2}\|\mathbf{x}-\mathbf{y}\|^{2},
\]
\[
f(\mathbf{x}) \leq f(\mathbf{y})+\nabla f(\mathbf{y})^{\top}(\mathbf{x}-\mathbf{y})
+\frac{L}{2}\|\mathbf{x}-\mathbf{y}\|^{2}.
\]
\end{claim}

\begin{figure}[ht]
    \centering
    \incfig{examples-of-lipschitz-class}
    \caption{Examples of Lipschitz classes. The solid line corresponds to the function.
    The dashed region is the space of $C_L^{0,0}$ (blue), and the dotted region is the space
    of $C_L^{1,1}$ (orange).}
    \label{fig:examples-of-lipschitz-class}
\end{figure}

\noindent How do we identify a Lipschitz class? One way to do it is using the definition,
but the following claim simplifies this process.
\begin{claim}
$f \in C_{L}^{k, m-1} \iff \|\nabla^{m} f(\mathbf{x})\| \leq L\quad \forall \mathbf{x}$.
\end{claim}
It can also be proved easily using the mean value theorem. Do it if you are an enthusiast!

\section{Finite differentiation}
Optimization algorithms often rely on gradients, which we sometimes compute analytically.
Still, many of us have been in the situation where a complicated derivative turns out to be
incorrect after spending 30 minutes (or more) deriving it. How can we check correctness
when there is no answer at the end of the textbook? Finite-difference methods provide a
simple sanity check; below we discuss several common schemes. At a point $\mathbf{x}$, we sample
a random direction $\mathbf{d}$ and compare the implemented directional derivative
$\nabla f(\mathbf{x})^{\top} \mathbf{d}$ with a numerical difference quotient in the same direction.
We denote
\[
  g:=\nabla f(\mathbf{x})^{\top}\mathbf{d},
\]
and approximate $g$ with step size $\varepsilon$ in floating-point arithmetic. We write
$\mathrm{fl}(\cdot)$ for computed quantities and use the model
\[
  |\mathrm{fl}(z)-z| \leq |z|\,\varepsilon_m.
\]
The key point is that we cannot simply take the smallest possible $\varepsilon$: in finite
differences too small step amplifies roundoff errors (subtractive-cancellation effects), so
$\varepsilon$ must balance truncation and roundoff errors.

\paragraph{Forward difference.}
The Taylor expansion gives
\[
  f(\mathbf{x}+\varepsilon \mathbf{d})=f(\mathbf{x})+g\,\varepsilon+O(\varepsilon^{2}),
\]
so
\[
  g=\frac{f(\mathbf{x}+\varepsilon \mathbf{d})-f(\mathbf{x})}{\varepsilon}+O(\varepsilon).
\]
Define the computed estimator
\[
  \widehat g_{\mathrm{fd}}(\varepsilon)=
  \mathrm{fl}\!\left(\frac{f(\mathbf{x}+\varepsilon \mathbf{d})-f(\mathbf{x})}{\varepsilon}\right).
\]
Then
\[
  |g-\widehat g_{\mathrm{fd}}(\varepsilon)|
  \le |g-\tfrac{f(\mathbf{x}+\varepsilon \mathbf{d})-f(\mathbf{x})}{\varepsilon}|
  +\left|\tfrac{f(\mathbf{x}+\varepsilon \mathbf{d})-f(\mathbf{x})}{\varepsilon}
  -\widehat g_{\mathrm{fd}}(\varepsilon)\right|.
\]
Assume $|f(\cdot)|\le L_0$ in a neighborhood of $\mathbf{x}$, and the truncation
error satisfies
\[
  \left|g-\tfrac{f(\mathbf{x}+\varepsilon \mathbf{d})-f(\mathbf{x})}{\varepsilon}\right|
  \le L_2\,\varepsilon.
\]
Then the roundoff term is bounded by
\[
  \left|\tfrac{f(\mathbf{x}+\varepsilon \mathbf{d})-f(\mathbf{x})}{\varepsilon}
  -\widehat g_{\mathrm{fd}}(\varepsilon)\right|
  \le \varepsilon_m\,\tfrac{|f(\mathbf{x}+\varepsilon \mathbf{d})|+|f(\mathbf{x})|}{\varepsilon}
  \le \tfrac{2L_0\,\varepsilon_m}{\varepsilon}.
\]
Therefore
\[
  |g-\widehat g_{\mathrm{fd}}(\varepsilon)|
  \le L_2\,\varepsilon+\tfrac{2L_0\,\varepsilon_m}{\varepsilon}.
\]
This shows that taking $\varepsilon$ too small increases the error due to
roundoff. Minimizing the bound gives
\[
  \varepsilon_{\mathrm{opt}}=
  \sqrt{\tfrac{2L_0\,\varepsilon_m}{L_2}}
  \approx \sqrt{\varepsilon_m}.
\]

\paragraph{Central difference.}
Using the expansions
\[
  f(\mathbf{x}+\varepsilon \mathbf{d})=f(\mathbf{x})+g\,\varepsilon
  +\tfrac{1}{2}\mathbf{d}^\top\nabla^2 f(\mathbf{x})\mathbf{d}\,\varepsilon^2+O(\varepsilon^3),
\]
\[
  f(\mathbf{x}-\varepsilon \mathbf{d})=f(\mathbf{x})-g\,\varepsilon
  +\tfrac{1}{2}\mathbf{d}^\top\nabla^2 f(\mathbf{x})\mathbf{d}\,\varepsilon^2+O(\varepsilon^3),
\]
we get
\[
  g=\frac{f(\mathbf{x}+\varepsilon \mathbf{d})-f(\mathbf{x}-\varepsilon \mathbf{d})}{2\varepsilon}
  +O(\varepsilon^2).
\]
With the computed estimator
\[
  \widehat g_{\mathrm{cd}}(\varepsilon)=
  \mathrm{fl}\!\left(\frac{f(\mathbf{x}+\varepsilon \mathbf{d})-f(\mathbf{x}-\varepsilon \mathbf{d})}
  {2\varepsilon}\right),
\]
and assuming $|f(\cdot)|\le L_0$ in the neighborhood of $\mathbf{x}$ and
a truncation bound $|O(\varepsilon^2)|\le L_3\varepsilon^2$, we obtain
\[
  |g-\widehat g_{\mathrm{cd}}(\varepsilon)|
  \le L_3\,\varepsilon^2+\tfrac{L_0\,\varepsilon_m}{\varepsilon}.
\]
Minimizing yields
\[
  \varepsilon_{\mathrm{opt}}=
  \sqrt[3]{\tfrac{L_0\,\varepsilon_m}{2L_3}}
  \approx \sqrt[3]{\varepsilon_m}.
\]

\paragraph{Complex step.}
If $f$ is an analytic function, then
\[
  f(\mathbf{x}+i\varepsilon \mathbf{d})=f(\mathbf{x})+i g\,\varepsilon+O(\varepsilon^2)+iO(\varepsilon^3),
\]
so
\[
  g=\frac{\operatorname{Im} f(\mathbf{x}+i\varepsilon \mathbf{d})}{\varepsilon}
  +O(\varepsilon^2).
\]
The previous schemes can be unstable because they subtract nearly equal numbers ($a-b$
with $a \approx b$) in floating-point arithmetic. This method avoids that subtraction, so we
can take $\varepsilon_{\mathrm{opt}}=\varepsilon_m$. This makes the scheme the most accurate.
